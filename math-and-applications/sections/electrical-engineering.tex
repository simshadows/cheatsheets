\subsection{DC Analysis Fundamentals}%
\label{sub:fundamentals}

\begin{multicols}{2}
    
    \CheatsheetEntryFrame{

        \CheatsheetEntryTitle{Current and Voltage}
        \begin{equation*}
            i \triangleq \frac{\diff{q}}{\diff{t}}
            \qquad
            v \triangleq \frac{\diff{w}}{\diff{q}}
        \end{equation*}

        \CheatsheetEntryTitle{Power}
        \begin{equation*}
            p \triangleq \frac{\diff{w}}{\diff{t}} = \frac{\diff{w}}{\diff{q}} \frac{\diff{q}}{\diff{t}}
            \qquad \Rightarrow \qquad
            p = vi
        \end{equation*}

        By \textit{passive sign convention}:
        \begin{itemize}
            \item power flowing \textit{into} a component is positive, and
            \item power flowing \textit{out} of a component is negative.
        \end{itemize}

    }

\end{multicols}
